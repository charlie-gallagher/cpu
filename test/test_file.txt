JMP   ; 00h
start

; The text file is merely a set of bytes to load into
; memory, so you don't need a 'load data' instruction
hello_world:
	'h'
	'e'
	'l'
	'l'
	'o'
	' '
	'w'
	'o'
	'r'
	'l'
	'd'
	00h  ; NULL terminator

; Print hello world
start:
	LDX_M
	FFh
	INCX
	00h	; no-op byte
	LDA_I
	hello_world
	STA_I
	IO_START
	BNE
	-8   ; Conditional jumps use relative addressing
	     ; The jump is relative to the next byte in memory

HLT
