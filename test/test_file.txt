JMP   ; 00h
start

; The text file is merely a set of bytes to load into
; memory, so you don't need a 'load data' instruction
hello_world:
	00h    ; Spacer to help with structure of loop
	'h'
	'e'
	'l'
	'l'
	'o'
	' '
	'w'
	'o'
	'r'
	'l'
	'd'
	00h  ; NULL terminator

; Print hello world
start:
	LDX_M
	hello_world
	INCX
	00h	; no-op byte
	LDA_I
	00h
	STA_I
	IO_START
	BNE
	-6   ; Conditional jumps use relative addressing
	     ; The jump is relative to the instruction, not
	     ; the operand

HLT
